#
# Errorbars with grid and logscale axis
#
rgb xgridgray, '0.9 0.9 0.9'
rgb ygridgray, '0.4 0.4 0.4'

plot {
   key hide

   xgrid {
      show
      line dashed, xgridgray, width=0.5
   }
   ygrid {
      show
      line dashed, ygridgray, width=0.5
   }

   xaxis {
      logscale
      title 'Time (s)'
      min 1e-7
      max 1e-2
      ticks {number 6}
   }
   yaxis {
      title 'Diffusivity ({\AA}$^2$/s)'
      min 2
      max 8
      ticks {number 4}
   }

   curve {
      read 'errorbars.dat', yerrcol=3
      line dotted, black
      symbol yerrbar, black
   }
}
